```typescript
// stores/batchStore.ts

import { create } from 'zustand'
import { Batch, BatchStatus } from '@/types/batch'

interface BatchState {
  batches: Batch[]
  selectedBatchId: string | null

  // Actions
  setBatches: (batches: Batch[]) => void
  updateBatchStatus: (batchId: string, status: BatchStatus) => void
  selectBatch: (batchId: string | null) => void
}

export const useBatchStore = create<BatchState>((set) => ({
  batches: [],
  selectedBatchId: null,

  setBatches: (batches) => set({ batches }),

  updateBatchStatus: (batchId, status) => set((state) => ({
    batches: state.batches.map(b =>
      b.id === batchId ? { ...b, ...status } : b
    )
  })),

  selectBatch: (batchId) => set({ selectedBatchId: batchId }),
}))
```

## 6. WebSocket í†µì‹ 

```typescript
// hooks/useWebSocket.ts

import { useEffect, useRef, useCallback } from 'react'
import { io, Socket } from 'socket.io-client'
import { useBatchStore } from '@/stores/batchStore'
import { useLogStore } from '@/stores/logStore'

export function useWebSocket() {
  const socketRef = useRef<Socket | null>(null)
  const updateBatchStatus = useBatchStore((s) => s.updateBatchStatus)
  const addLog = useLogStore((s) => s.addLog)

  useEffect(() => {
    // Connect to Station Service WebSocket
    socketRef.current = io('/', {
      transports: ['websocket'],
    })

    const socket = socketRef.current

    // Event handlers
    socket.on('batch_status', (data) => {
      updateBatchStatus(data.batch_id, data)
    })

    socket.on('step_start', (data) => {
      updateBatchStatus(data.batch_id, {
        currentStep: data.step,
        stepIndex: data.index,
      })
    })

    socket.on('step_complete', (data) => {
      updateBatchStatus(data.batch_id, {
        completedSteps: data.index + 1,
      })
    })

    socket.on('sequence_complete', (data) => {
      updateBatchStatus(data.batch_id, {
        status: data.result.overall_pass ? 'completed' : 'error',
        result: data.result,
      })
    })

    socket.on('log', (data) => {
      addLog({
        batchId: data.batch_id,
        level: data.level,
        message: data.message,
        timestamp: new Date(),
      })
    })

    return () => {
      socket.disconnect()
    }
  }, [])

  const subscribe = useCallback((batchIds: string[]) => {
    socketRef.current?.emit('subscribe', { batch_ids: batchIds })
  }, [])

  return { subscribe }
}
```
